		ROOT = {

			}

			has_country_modifier = mercs_bought

		}

   }

								which = geckomercFL

			}

						which = geckomercvalue
						value = trigger_value:num_of_mercenaries
						who = ROOT
			}
	}

	option = {
	}

	id = gecko.28
	title = gecko.28.title
	desc = gecko.28.desc
	picture = LAND_MILITARY_eventPicture

	is_triggered_only = yes

	option = {
		name = gecko.28.a
		
		custom_tooltip = gecko_increased_merc_limit_tt
			
			hidden_effect = {
				export_to_variable = {
						which = geckomercvalue
						value = trigger_value:num_of_mercenaries
						who = ROOT
					}
				#add_country_modifier = {
				#	name = "We_stand_alone"
				#	duration = 3650
				#}
				if = {
					limit = {
						OR = {
							AND = {
								 has_country_modifier = We_stand_alone
								 has_age_ability = ab_hesse_mercs
							}
							AND = {
								 has_country_modifier = We_stand_alone
								 has_age_ability = ab_burgundy_merc_manpower
							}
						}
					}
					set_variable = {            
								which = availablegeckomercs
								value = 1.3
						}	
					add_country_modifier = {
						name = "130percentavailablemercs"
						duration = -1
						hidden = yes
					}	
				}
				else = {
					set_variable = {            
							which = availablegeckomercs
							value = 1.15
					}
				}	
			}
	}
	option = {
		name = gecko.28.b
		
		custom_tooltip = gecko_decreased_merc_limit_tt
			
			hidden_effect = {
				export_to_variable = {
						which = geckomercvalue
						value = trigger_value:num_of_mercenaries
						who = ROOT
					}
				add_country_modifier = {
					name = "increased_merc_quality"
					duration = 3650
				}
				if = {
					limit = {
						OR = {
							AND = {
								 has_country_modifier = increased_merc_quality
								 has_age_ability = ab_hesse_mercs

								 has_country_modifier = increased_merc_quality
								 has_age_ability = ab_burgundy_merc_manpower
							}
						}
					}
					set_variable = {            
								which = availablegeckomercs
								value = 1
						}
					add_country_modifier = {
						name = "100percentavailablemercs"
						duration = -1
						hidden = yes
					}
				}
				else = {
					set_variable = {            
							which = availablegeckomercs
							value = 0.85
					}

			}
